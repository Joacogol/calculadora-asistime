function calcularAhorro() {

    // Obtener valores ingresados

    let leadsDiarios = parseInt(document.getElementById("leadsDiarios").value);

    let costoHora = parseFloat(document.getElementById("costoHora").value);

    let tiempoPorLead = 15; // Minutos por lead

    let diasLaborales = 22;


    // Calcular tiempo perdido sin IA

    let tiempoDiario = (leadsDiarios * tiempoPorLead) / 60; // en horas

    let tiempoMensual = tiempoDiario * diasLaborales; // en horas

    

    // Calcular costo mensual sin IA

    let costoMensual = tiempoMensual * costoHora;


    // Mostrar resultados

    document.getElementById("resultado").innerHTML = `

        <p><strong>Tiempo perdido sin IA:</strong> ${tiempoMensual.toFixed(2)} horas/mes</p>

        <p><strong>Costo mensual en sueldos por calificación manual:</strong> $${costoMensual.toFixed(2)} UYU</p>

        <p><strong>Tiempo ahorrado con Asistime:</strong> ${tiempoMensual.toFixed(2)} horas/mes</p>

        <p><strong>Dinero ahorrado con Asistime:</strong> $${costoMensual.toFixed(2)} UYU/mes</p>

    `;

}


// HTML para la calculadora

document.write(`

    <div style='max-width: 400px; margin: auto; font-family: Arial, sans-serif;'>

        <h2>Calculadora de Ahorro con IA</h2>

        <label>Cantidad de leads diarios:</label>

        <input type="number" id="leadsDiarios" value="40" style='width: 100%; padding: 5px; margin-bottom: 10px;'><br>

        

        <label>Costo por hora del comercial (UYU):</label>

        <input type="number" id="costoHora" value="500" style='width: 100%; padding: 5px; margin-bottom: 10px;'><br>

        

        <button onclick="calcularAhorro()" style='padding: 10px; background-color: blue; color: white; border: none; cursor: pointer; width: 100%;'>

            Calcular Ahorro

        </button>

        

        <div id="resultado" style='margin-top: 20px;'></div>

    </div>

`);
